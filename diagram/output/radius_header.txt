protocol name: RadiusAttribute;
radius_header
pragma pack(push, 1)
	struct radius_header
	{
		/** RADIUS message code */
		uint8_t code;
		/** RADIUS message ID */
		uint8_t id;
		/** RADIUS message length */
		uint16_t length;
		/** Used to authenticate the reply from the RADIUS server and to encrypt passwords */
		uint8_t authenticator[16];
	};
#pragma pack
